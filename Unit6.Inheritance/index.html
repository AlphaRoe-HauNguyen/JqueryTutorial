Javascript - Inheritance

function Monkey() {
    this.hands = 2;
    this.legs = 2;
}
Monkey.prototype.walk = function () { alert('Monkey is walking'); }

function Human() { }
Human.prototype.talk = function () { alert('Human is talking'); }
Human.prototype.think = function () { alert('Human is thinking'); }
Human.prototype.__proto__ = new Monkey();

var human = new Human();
alert(human.hands);
human.walk();
human.think();
